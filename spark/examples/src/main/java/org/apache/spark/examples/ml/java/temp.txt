public Rating() {}

    public void Als_edge(String invoice_num, String product_code) {
      this.invoice_num = invoice_num;
      this.product_code = product_code;
    }

    public static Rating parseRating(String str) {
      String[] fields = str.split("::");
      if (fields.length != 2) {
        throw new IllegalArgumentException("Each line must contain 2 fields");
      }
      int invoice_num = Integer.parseInt(fields[0]);
      int product_code = Integer.parseInt(fields[1]);
      //float rating = Float.parseFloat(fields[2]);
      return new Rating(invoice_num, product_code);
    }
  }
